package main

import (
	"fmt"
	"log"
	"net"
)

const (
	website string = "www.dirtgrub.000webhostapp.com:80"
)

func main() {
	_, err := net.Dial("udp", website)
	if err != nil {
		log.Fatal("DirtGrub is offline or just taking a nap ", err)
	}
	fmt.Print("DirtGrub is fully operational")

}
